# CORS / CSRF — allow the Vercel frontend to call admin.musicconnectz.net.
# (Requires django-cors-headers in INSTALLED_APPS + middleware.)
#
# IMPORTANT — the deploy error you hit:
#   corsheaders.E013: Origin '://musicconnectz.net' is missing scheme or netloc
# means somewhere an origin was added WITHOUT its scheme (e.g. "musicconnectz.net"
# or a stray "://musicconnectz.net"). corsheaders rejects those at startup.
# The helper below normalizes every origin (adds https:// if missing, drops blanks),
# so a bad value can never crash the boot again. REPLACE your existing
# CORS_ALLOWED_ORIGINS / CSRF_TRUSTED_ORIGINS blocks with this.
from urllib.parse import urlparse


def _origins(*values):
    out = []
    for v in values:
        v = (v or "").strip().rstrip("/")
        if not v:
            continue
        if "://" not in v:
            v = "https://" + v            # bare host -> https://host
        p = urlparse(v)
        if p.scheme and p.netloc:         # only keep well-formed origins
            origin = f"{p.scheme}://{p.netloc}"
            if origin not in out:
                out.append(origin)
    return out


# Optionally add more via env: CORS_EXTRA_ORIGINS="https://a.com,https://b.com"
_EXTRA = os.environ.get("CORS_EXTRA_ORIGINS", "").split(",")

CORS_ALLOWED_ORIGINS = _origins(
    "https://musicconnectz.net",
    "https://www.musicconnectz.net",
    "https://ctkoth-music-connectz-frontend-wy3x.vercel.app",
    "http://localhost:5173",
    *_EXTRA,
)
CSRF_TRUSTED_ORIGINS = _origins(
    "https://musicconnectz.net",
    "https://www.musicconnectz.net",
    "https://admin.musicconnectz.net",
    "https://ctkoth-music-connectz-frontend-wy3x.vercel.app",
    *_EXTRA,
)
